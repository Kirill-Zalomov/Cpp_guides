1) Установка cryptopp:
sudo apt install -y libcrypto++-dev


2) Просмотр файлов пакета:
apt-file show libcrypto++-dev


3) Добавление в CMake:

```
find_path(CRYPTOPP_INCLUDE_DIR
	cryptopp
    PATHS
    	/usr/include
)

find_library(CRYPTOPP_LIBRARY
    NAMES
    	cryptopp
    	crypto++
    PATHS
    	/usr/lib/x86_64-linux-gnu/
)

if (CRYPTOPP_INCLUDE_DIR AND CRYPTOPP_LIBRARY)
    message(STATUS "Crypto++ found: ${CRYPTOPP_LIBRARY}")
else()
    message(FATAL_ERROR "Crypto++ not found!")
endif()


add_executable(prog main.cpp)
target_include_directories(prog
	PRIVATE
		${CRYPTOPP_INCLUDE_DIR}
)
target_link_libraries(prog
	PRIVATE
		${CRYPTOPP_LIBRARY}
)
```
